  function setQuickFilter(range) {
     const startInput = document.getElementById('report-start');
     const endInput = document.getElementById('report-end');
     const checkAll = document.getElementById('check-all-time');
     
     checkAll.checked = false; 
     toggleAllTime();
     
     const now = new Date();
     let start = new Date();
     let end = new Date();
     
     if (range === 'today') {
        // Start and End depend on whether we want to capture "just today"
        start = now; end = now;
     } else if (range === 'yesterday') {
        start.setDate(now.getDate() - 1);
        end.setDate(now.getDate() - 1);
     } else if (range === 'last7') {
        start.setDate(now.getDate() - 6);
        end = now;
     } else if (range === 'thisMonth') {
        start = new Date(now.getFullYear(), now.getMonth(), 1);
        end = new Date(now.getFullYear(), now.getMonth() + 1, 0);
     } else if (range === 'lastMonth') {
        start = new Date(now.getFullYear(), now.getMonth() - 1, 1);
        end = new Date(now.getFullYear(), now.getMonth(), 0);
     }
     
     startInput.value = start.toISOString().split('T')[0];
     endInput.value = end.toISOString().split('T')[0];
     
     generateReport();
  }

  function downloadCSV() {
    if (!currentReportData) return showToast("No report data to download", "error");
    
    let csvContent = "data:text/csv;charset=utf-8,";
    var filename = 'Report_' + currentReportType + '_' + new Date().toISOString().split('T')[0] + '.csv';

    if (currentReportType === 'expense') {
        csvContent += "Date,Item,Category,For Whom,Amount,Receipt\n";
        currentReportData.topItems.forEach(row => {
            var rowStr = [
                row.date ? new Date(row.date).toLocaleDateString() : "",
                '"' + (row.item || '').replace(/"/g, '""') + '"',
                '"' + (row.cat || '').replace(/"/g, '""') + '"',
                '"' + (row.whom || '').replace(/"/g, '""') + '"',
                row.amount,
                row.receipt || ""
            ].join(",");
            csvContent += rowStr + "\n";
        });
    } else {
        csvContent += "Date,Person,Action,Amount,Net Effect,Receipt\n";
        currentReportData.activityLog.forEach(row => {
            var rowStr = [
                row.date ? new Date(row.date).toLocaleDateString() : "",
                '"' + (row.person || '').replace(/"/g, '""') + '"',
                '"' + (row.type || '').replace(/"/g, '""') + '"',
                row.amount,
                row.direction,
                row.receipt || ""
            ].join(",");
            csvContent += rowStr + "\n";
        });
    }

    const encodedUri = encodeURI(csvContent);
    const link = document.createElement("a");
    link.setAttribute("href", encodedUri);
    link.setAttribute("download", filename);
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  }

  async function readInputFile(id) {
    const fileInput = document.getElementById(id);
    if (!fileInput || !fileInput.files.length) return null;
    const file = fileInput.files[0];
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = (e) => {
        const content = e.target.result.split(',')[1];
        resolve({
          fileName: file.name,
          mimeType: file.type,
          fileData: content
        });
      };
      reader.onerror = error => reject(error);
      reader.readAsDataURL(file);
    });
  }

  function deleteOption(type, value) {
      if(!confirm('Delete option "' + value + '"?\n\nThis will remove "' + value + '" from ALL existing transaction records efficiently.\n\nAre you sure you want to proceed?')) return;
      
      showToast("Removing option...", "info");
      
      google.script.run
        .withSuccessHandler((res) => {
            if(res && res.error) {
                showToast("Error: " + res.message, "error");
            } else {
                showToast('Removed "' + value + '" from ' + res.count + ' records.', 'success');
                loadSuggestions(); // Reload to refresh lists
                // close the dropdown
                const options = document.getElementById(type + '-options');
                if(options) options.classList.add('hidden');
            }
        })
        .withFailureHandler(err => showToast("Failed: " + err.message, "error"))
        .deleteOptionMetadata(type, value);
  }

  function addCustomFilterOption(type) {
    const input = document.getElementById(type + '-custom-input');
    const value = input.value.trim();
    if (!value) return;
    
    const optionsContainer = document.getElementById(type + '-options');
    const customDiv = optionsContainer.querySelector('.multi-select-custom-input');
    
    // Check if value already exists
    const existingCheckbox = optionsContainer.querySelector('input[value="' + value + '"]');
    if (existingCheckbox) {
      existingCheckbox.checked = true;
      handleMultiSelectChange(type, existingCheckbox);
      input.value = '';
      return;
    }
    
    // Create new option and insert before the custom input div
    // Use same structure as main list
    const opt = document.createElement('div');
    opt.className = 'multi-select-option flex items-center justify-between group';
      
    const labelWrap = document.createElement('label');
    labelWrap.className = 'flex-1 flex items-center gap-2 cursor-pointer';
    labelWrap.innerHTML = '<input type="checkbox" value="' + value + '" checked onchange="handleMultiSelectChange(\'' + type + '\', this)"> ' + value;
    
    const delBtn = document.createElement('button');
    delBtn.className = 'text-gray-300 hover:text-red-500 p-1 opacity-0 group-hover:opacity-100 transition-opacity';
    delBtn.innerHTML = '<i class="fas fa-trash-alt text-[10px]"></i>';
    delBtn.title = "Delete this option from records";
    delBtn.onclick = (e) => {
          e.preventDefault(); 
          deleteOption(type, value);
    };

    opt.appendChild(labelWrap);
    opt.appendChild(delBtn);

    optionsContainer.insertBefore(opt, customDiv);
    
    // Trigger change
    const newCheckbox = opt.querySelector('input');
    handleMultiSelectChange(type, newCheckbox);
    
    // Clear input
    input.value = '';
  }

  function toggleMultiSelect(type) {
    const options = document.getElementById(type + '-options');
    const trigger = options.previousElementSibling;
    const isHidden = options.classList.contains('hidden');
    
    // Close all other dropdowns first
    document.querySelectorAll('.multi-select-options').forEach(el => {
      el.classList.add('hidden');
      el.previousElementSibling.classList.remove('open');
    });
    
    if (isHidden) {
      options.classList.remove('hidden');
      trigger.classList.add('open');
    }
  }

  function handleMultiSelectChange(type, checkbox) {
    const optionsContainer = document.getElementById(type + '-options');
    const allCheckbox = optionsContainer.querySelector('input[value="All"]');
    const otherCheckboxes = optionsContainer.querySelectorAll('input:not([value="All"])');
    
    if (checkbox.value === 'All') {
      // If "All" is checked, uncheck others
      if (checkbox.checked) {
        otherCheckboxes.forEach(cb => cb.checked = false);
      }
    } else {
      // If any specific item is checked, uncheck "All"
      if (checkbox.checked) {
        allCheckbox.checked = false;
      }
      // If no specific items checked, check "All"
      const anyChecked = Array.from(otherCheckboxes).some(cb => cb.checked);
      if (!anyChecked) {
        allCheckbox.checked = true;
      }
    }
    
    updateMultiSelectText(type);
  }

  function selectAllMulti(type) {
    const optionsContainer = document.getElementById(type + '-options');
    const allCheckbox = optionsContainer.querySelector('input[value="All"]');
    const otherCheckboxes = optionsContainer.querySelectorAll('input:not([value="All"])');
    otherCheckboxes.forEach(cb => cb.checked = true);
    if(allCheckbox) allCheckbox.checked = false;
    updateMultiSelectText(type);
  }

  function clearAllMulti(type) {
    const optionsContainer = document.getElementById(type + '-options');
    const allCheckbox = optionsContainer.querySelector('input[value="All"]');
    const otherCheckboxes = optionsContainer.querySelectorAll('input:not([value="All"])');
    otherCheckboxes.forEach(cb => cb.checked = false);
    if(allCheckbox) allCheckbox.checked = true; // Fallback to ALL if cleared
    updateMultiSelectText(type);
  }

    const textSpan = document.getElementById(type + '-selected-text');
    const allCheckbox = optionsContainer.querySelector('input[value="All"]');
    
    if (allCheckbox && allCheckbox.checked) {
      const allLabels = { cat: 'All Categories', whom: 'All', exptype: 'All Types', person: 'All' };
      textSpan.innerText = allLabels[type] || 'All';
      return;
    }
    
    const checkedLabels = Array.from(optionsContainer.querySelectorAll('input:checked'))
      .map(cb => cb.value)
      .filter(v => v !== 'All');
    
    if (checkedLabels.length === 0) {
      textSpan.innerText = 'None selected';
    } else if (checkedLabels.length <= 2) {
      textSpan.innerText = checkedLabels.join(', ');
    } else {
      textSpan.innerText = checkedLabels.length + ' selected';
    }
  }

  function getMultiSelectValues(type) {
    const optionsContainer = document.getElementById(type + '-options');
    if (!optionsContainer) return ['All'];
    
    const allCheckbox = optionsContainer.querySelector('input[value="All"]');
    if (allCheckbox && allCheckbox.checked) {
      return ['All'];
    }
    
    const selected = Array.from(optionsContainer.querySelectorAll('input:checked'))
      .map(cb => cb.value)
      .filter(v => v !== 'All');
    
    return selected.length > 0 ? selected : ['All'];
  }

  // --- SUBMIT HANDLER ---
  async function handleFormSubmit(e) {
    e.preventDefault();
    const form = e.target;
    const btn = form.querySelector('button[type="submit"]');
    
    if (!btn.hasAttribute('data-original-html')) {
        btn.setAttribute('data-original-html', btn.innerHTML);
    }
    const originalContent = btn.getAttribute('data-original-html');
    
    const isEdit = form.querySelector('input[name="rowIndex"]').value !== "";
    const type = form.querySelector('input[name="formType"]').value;
    const fileInputId = type === 'expense' ? 'exp-file' : 'lend-file';

    btn.innerHTML = '<i class="fas fa-circle-notch fa-spin"></i> Processing...';
    btn.disabled = true;
    
    // SAFETY TIMEOUT INCREASED TO 60 SECONDS
    const safetyTimer = setTimeout(() => {
        if(btn.disabled) {
            btn.innerHTML = originalContent;
            btn.disabled = false;
            showToast("Server timeout. Operation may still complete.", 'error');
        }
    }, 60000); 

    try {
      const formData = {};
      new FormData(form).forEach((value, key) => formData[key] = value);
      
      const fileObj = await readInputFile(fileInputId);
      if (fileObj) {
        formData.fileData = fileObj.fileData;
        formData.fileName = fileObj.fileName;
        formData.mimeType = fileObj.mimeType;
      }
      
      const runner = google.script.run
        .withSuccessHandler((res) => {
          clearTimeout(safetyTimer);
          btn.innerHTML = originalContent;
          btn.disabled = false;
          
          if (res && res.success) {
            showToast(isEdit ? 'Updated Successfully!' : 'Entry Saved!', 'success');
            cancelEdit(formData.formType); 
            loadDashboard(); 
            loadSuggestions();
            if (!document.getElementById('view-reports').classList.contains('hidden')) generateReport();
          } else {
            showToast('Error: ' + (res ? res.message : 'Unknown error'), 'error');
          }
        })
        .withFailureHandler(err => {
           clearTimeout(safetyTimer);
           btn.innerHTML = originalContent;
           btn.disabled = false;
           showToast("Connection Error: " + err.message, 'error');
        });

      if (isEdit) {
        runner.updateTransaction(formData);
      } else {
        runner.saveData(formData);
      }

    } catch (err) {
      clearTimeout(safetyTimer);
      btn.innerHTML = originalContent;
      btn.disabled = false;
      showToast("Client Error: " + err.message, 'error');
    }
  }

  function deleteRecord(sheetType, rowIndex) {
    if(confirm('Delete this entry?')) {
      document.getElementById('loader').classList.remove('hidden');
      google.script.run
        .withSuccessHandler(res => {
           document.getElementById('loader').classList.add('hidden');
           if(res.success) { 
             showToast('Deleted Successfully', 'success'); 
             loadDashboard(); 
             if(!document.getElementById('view-reports').classList.contains('hidden')) generateReport(); 
           }
           else showToast('Failed: ' + res.message, 'error');
        })
        .withFailureHandler(err => {
           document.getElementById('loader').classList.add('hidden');
           showToast('Connection Error: ' + err.message, 'error');
        })
        .deleteTransaction(sheetType, rowIndex);
    }
  }

  function switchTab(tab) {
    document.querySelectorAll('.view-section').forEach(el => el.classList.add('hidden'));
    document.querySelectorAll('.tab-btn').forEach(el => { 
        // Thoroughly reset button states, including hard-coded gradients from HTML
        el.classList.remove('active', 'text-white', 'text-slate-800', 'shadow-md', 'shadow-lg', 'bg-gradient-to-r', 'from-slate-800', 'to-slate-900', 'bg-white'); 
        el.classList.add('text-gray-500', 'bg-transparent'); 
        el.style.backgroundColor = '';
        el.style.borderColor = 'transparent';
    });
    
    // Set active state using custom style and standard classes
    const activeBtn = document.getElementById('tab-' + tab);
    if(activeBtn) {
      activeBtn.classList.remove('text-gray-500', 'bg-transparent');
      activeBtn.classList.add('active', 'text-white', 'shadow-lg');
      activeBtn.style.backgroundColor = 'var(--accent-color)';
      activeBtn.style.borderColor = 'var(--accent-color)';
    }
    
    document.getElementById('view-' + tab).classList.remove('hidden');
    
    // Set Default Date if empty
    const todayStr = new Date().toISOString().split('T')[0];
    if (tab === 'expense') {
        const d = document.getElementById('exp-date');
        if(!d.value) d.value = todayStr;
    } else if (tab === 'lending') {
        const d = document.getElementById('lend-date');
        if(!d.value) d.value = todayStr;
    }

    if (tab === 'reports') setReportType(currentReportType);
    updateHeaderStats(tab);
  }

  function loadDashboard() {
    google.script.run.withSuccessHandler(data => {
      if(data.error) return;
      globalData = data;
      globalSearchIndex = data.searchIndex || [];
      let activeTab = 'expense';
      if (!document.getElementById('view-lending').classList.contains('hidden')) activeTab = 'lending';
      if (!document.getElementById('view-reports').classList.contains('hidden')) activeTab = 'reports';
      updateHeaderStats(activeTab);
      
      const renderRow = (row) => {
        let dateStr = new Date(row.date).toLocaleDateString(undefined, {month:'short', day:'numeric'});
        let colorClass = 'text-slate-800';
        let prefix = '-';
        if(row.isLend) {
           let isMoneyIn = row.cat.includes('Borrowed') || row.cat.includes('Received') || row.cat.includes('Return (In)');
           colorClass = isMoneyIn ? 'text-emerald-600' : 'text-rose-600';
           prefix = isMoneyIn ? '+' : '-';
        }
        
        // Attachment thumbnail preview
        let attachmentThumb = '';
        if (row.receipt) {
           const isImage = row.receipt.match(/\.(jpg|jpeg|png|gif|webp)/i) || row.receipt.includes('image') || !row.receipt.includes('pdf');
           if (isImage) {
              attachmentThumb = '<div class="attachment-thumb" onclick="event.stopPropagation(); openTransactionModal(\'' + encodeURIComponent(JSON.stringify(row)).replace(/'/g, "%27") + '\');"><img src="' + row.receipt + '" alt="Receipt"></div>';
           } else {
              attachmentThumb = '<div class="attachment-thumb" onclick="event.stopPropagation(); window.open(\'' + row.receipt + '\', \'_blank\');"><i class="fas fa-file-pdf pdf-icon"></i></div>';
           }
        }

        const dataStr = encodeURIComponent(JSON.stringify(row)).replace(/'/g, "%27");
        var defaultIcon = '<div class="h-10 w-10 rounded-full bg-slate-50 flex items-center justify-center text-slate-400"><i class="fas ' + (row.isLend ? 'fa-hand-holding-usd' : 'fa-receipt') + ' text-sm"></i></div>';
        return '<li class="flex justify-between items-center bg-white p-4 rounded-xl border border-gray-100 shadow-sm relative group cursor-pointer hover:bg-gray-50 transition-colors" onclick="openTransactionModal(\'' + dataStr + '\')">' +
             '<div class="flex items-center gap-3">' +
               (attachmentThumb || defaultIcon) +
               '<div><div class="font-bold text-slate-700 text-sm">' + row.cat + ' <span class="text-slate-400 font-normal">(' + row.whom + ')</span></div><div class="text-[10px] text-gray-400">' + dateStr + ' &bull; ' + row.detail + '</div></div>' +
             '</div>' +
             '<div class="flex flex-col items-end gap-1">' +
                '<div class="font-bold ' + colorClass + '">' + prefix + appSettings.currency + row.amt.toLocaleString() + '</div>' +
                '<div class="flex items-center gap-1 opacity-0 group-hover:opacity-100 transition-opacity">' +
                   '<button onclick="event.stopPropagation(); editRecord(\'' + dataStr + '\')" class="text-gray-300 hover:text-amber-500 p-1" title="Edit"><i class="fas fa-edit text-xs"></i></button>' +
                   '<button onclick="event.stopPropagation(); deleteRecord(\'' + row.sheet + '\', ' + row.row + ')" class="text-gray-300 hover:text-red-500 p-1" title="Delete"><i class="fas fa-trash-alt text-xs"></i></button>' +
                '</div>' +
             '</div>' +
          '</li>';
      };

      const expList = document.getElementById('recent-list-expense');
      if (expList) {
        expList.innerHTML = '';
        if(data.recentExpenses.length === 0) expList.innerHTML = '<li class="text-center text-xs text-gray-300">No recent expenses</li>';
        data.recentExpenses.forEach(row => expList.innerHTML += renderRow(row));
      }
      const lendList = document.getElementById('recent-list-lending');
      if (lendList) {
        lendList.innerHTML = '';
        if(data.recentLending.length === 0) lendList.innerHTML = '<li class="text-center text-xs text-gray-300">No recent transactions</li>';
        data.recentLending.forEach(row => lendList.innerHTML += renderRow(row));
      }
    }).getDashboardData();
  }

  function updateFloatingLabelStates() {
    document.querySelectorAll('.floating-input').forEach(input => {

  // --- AUTOCOMPLETE FUNCTIONALITY ---
  let autocompleteData = {};
  let activeAutocomplete = null;

  function setupAutocomplete(inputId, dataKey) {
    const input = document.getElementById(inputId);
    if (!input) return;

    // Create dropdown if it doesn't exist
    let dropdown = document.getElementById(inputId + '-autocomplete');
    if (!dropdown) {
      dropdown = document.createElement('div');
      dropdown.id = inputId + '-autocomplete';
      dropdown.className = 'autocomplete-dropdown hidden';
      input.parentElement.style.position = 'relative';
      input.parentElement.appendChild(dropdown);
    }

    // Input event listener
    input.addEventListener('input', function(e) {
      const value = this.value.toLowerCase();
      const suggestions = autocompleteData[dataKey] || [];
      
      if (value.length === 0) {
        dropdown.classList.add('hidden');
        return;
      }

      const filtered = suggestions.filter(item => 
        item.toLowerCase().includes(value)
      ).slice(0, 8); // Limit to 8 suggestions

      if (filtered.length === 0) {
        dropdown.classList.add('hidden');
        return;
      }

      dropdown.innerHTML = filtered.map(function(item) {
        return '<div class="autocomplete-item" data-value="' + item + '">' + item + '</div>';
      }).join('');
      
      dropdown.classList.remove('hidden');
      activeAutocomplete = { input, dropdown };

      // Add click listeners to items
      dropdown.querySelectorAll('.autocomplete-item').forEach(item => {
        item.addEventListener('click', function() {
          input.value = this.dataset.value;
          dropdown.classList.add('hidden');
          updateFloatingLabelStates();
        });
      });
    });

    // Close on blur (with delay for click handling)
    input.addEventListener('blur', function() {
      setTimeout(() => {
        if (dropdown) dropdown.classList.add('hidden');
      }, 150);
    });

    // Keyboard navigation
    input.addEventListener('keydown', function(e) {
      if (dropdown.classList.contains('hidden')) return;
      
      const items = dropdown.querySelectorAll('.autocomplete-item');
      const current = dropdown.querySelector('.autocomplete-item.highlighted');
      let index = current ? Array.from(items).indexOf(current) : -1;

      if (e.key === 'ArrowDown') {
        e.preventDefault();
        index = Math.min(index + 1, items.length - 1);
      } else if (e.key === 'ArrowUp') {
        e.preventDefault();
        index = Math.max(index - 1, 0);
      } else if (e.key === 'Enter' && current) {
        e.preventDefault();
        current.click();
        return;
      } else if (e.key === 'Escape') {
        dropdown.classList.add('hidden');
        return;
      } else {
        return;
      }

      items.forEach(item => item.classList.remove('highlighted'));
      if (items[index]) {
        items[index].classList.add('highlighted');
        items[index].scrollIntoView({ block: 'nearest' });
      }
    });
  }

  // --- GLOBAL SEARCH ---
  function setupGlobalSearch() {
     const input = document.getElementById('global-search');
     const results = document.getElementById('search-results-dropdown');
     
     if(!input || !results) return;

     input.addEventListener('input', (e) => {
        const query = e.target.value.toLowerCase().trim();
        if(query.length < 2) {
           results.classList.add('hidden');
           return;
        }
        
        // Google-like fuzzy search: split query into words and match ANY word
        const queryWords = query.split(/\s+/).filter(w => w.length >= 1);
        const scoredMatches = globalSearchIndex
          .map(item => {
            if (!item.searchStr) return null;
            const searchStr = item.searchStr.toLowerCase();
            const searchWords = searchStr.split(/\s+/);
            let score = 0;
            
            queryWords.forEach(qWord => {
              // Exact word match (highest priority)
              if (searchWords.some(sw => sw === qWord)) {
                score += 10;
              }
              // Word starts with query word (high priority)
              else if (searchWords.some(sw => sw.startsWith(qWord))) {
                score += 5;
              }
              // Word contains query word (medium priority)
              else if (searchStr.includes(qWord)) {
                score += 3;
              }
              // Partial character sequence match (lower priority)
              else {
                for (let sw of searchWords) {
                  if (sw.includes(qWord.substring(0, Math.min(3, qWord.length)))) {
                    score += 1;
                    break;
                  }
                }
              }
            });
            
            return score > 0 ? { ...item, score } : null;
          })
          .filter(Boolean)
          .sort((a, b) => b.score - a.score)
          .slice(0, 50);
        
        const matches = scoredMatches;
        
        if(matches.length === 0) {
           results.innerHTML = '<div class="p-4 text-center text-gray-400 text-xs">No matching transactions found</div>';
        } else {
           let html = '';
           matches.forEach(item => {
               const dataStr = encodeURIComponent(JSON.stringify(item)).replace(/'/g, "%27");
               let iconClass = item.isLend ? 'fa-hand-holding-usd text-emerald-500' : 'fa-receipt text-indigo-500';
               let amtColor = item.isLend ? (item.amount > 0 ? 'text-emerald-600' : 'text-rose-600') : 'text-slate-700';
               
               let displayDate = new Date(item.date).toLocaleDateString([], {day:'numeric', month:'short', year:'2-digit'});
               
               // Attachment thumbnail for search results  
               let thumbHtml = '<div class="h-8 w-8 rounded-full bg-gray-100 flex items-center justify-center"><i class="fas ' + iconClass + ' text-xs"></i></div>';
               if (item.receipt) {
                  const isImage = item.receipt.match(/\.(jpg|jpeg|png|gif|webp)/i) || item.receipt.includes('image') || !item.receipt.includes('pdf');
                  if (isImage) {
                     thumbHtml = '<div class="attachment-thumb" style="width:2rem;height:2rem;"><img src="' + item.receipt + '" alt=""></div>';
                  } else {
                     thumbHtml = '<div class="attachment-thumb" style="width:2rem;height:2rem;"><i class="fas fa-file-pdf pdf-icon text-xs"></i></div>';
                  }
               }
               
               html += '<div onclick="openTransactionModal(\'' + dataStr + '\'); document.getElementById(\'search-results-dropdown\').classList.add(\'hidden\');" class="p-3 hover:bg-gray-50 cursor-pointer flex justify-between items-center group transition-colors">' +
                  '<div class="flex items-center gap-3">' +
                     thumbHtml +
                     '<div>' +
                        '<div class="text-xs font-bold text-slate-700">' + (item.cat || item.type) + ' <span class="font-normal text-gray-500">(' + item.whom + ')</span></div>' +
                        '<div class="text-[10px] text-gray-400">' + displayDate + ' &bull; ' + (item.detail||'No details') + '</div>' +
                     '</div>' +
                  '</div>' +
                  '<div class="text-right">' +
                     '<div class="text-sm font-bold ' + amtColor + '">' + appSettings.currency + item.amt.toLocaleString() + '</div>' +

  // --- EXPAND/COLLAPSE TOGGLE ---
  function toggleExpand(id) {
    const toggle = document.querySelector('[data-toggle="' + id + '"]');
    const content = document.getElementById(id);
    
    if (toggle && content) {
      const isOpening = content.classList.contains('hidden');
      
      if (isOpening) {
        // Accordion behavior: auto-close other expanded sections in the report
        document.querySelectorAll('.collapsible-content.expanded').forEach(other => {
          if (other.id !== id) {
            other.classList.add('hidden');
            other.classList.remove('expanded');
            const otherToggle = document.querySelector('[data-toggle="' + other.id + '"]');
            if (otherToggle) {
              otherToggle.classList.remove('expanded');
              const otherIcon = otherToggle.querySelector('.fa-minus, .fa-chevron-down');
              if (otherIcon) {
                if(otherIcon.classList.contains('fa-minus')) { otherIcon.classList.remove('fa-minus'); otherIcon.classList.add('fa-plus'); }
              }
            }
          }
        });
      }

      toggle.classList.toggle('expanded');
      content.classList.toggle('expanded');
      content.classList.toggle('hidden'); // Use standard hidden class from renderReport
      
      const icon = toggle.querySelector('.fa-plus, .fa-minus');
      if(icon) {
         if(content.classList.contains('hidden')) {
             icon.classList.remove('fa-minus'); icon.classList.add('fa-plus');
         } else {
             icon.classList.remove('fa-plus'); icon.classList.add('fa-minus');
         }
      }
    }
  }

   // --- BANK SEARCH MODULE ---
   let bankDataCache = null;
   let currentBankMatches = []; // Store current filtered results
   let bankDisplayLimit = 50;   // Items to show initially
   const BANK_PAGE_SIZE = 50;

   function openBankSearch() {
     document.getElementById('bank-search-modal').classList.remove('hidden');
     setTimeout(() => document.getElementById('bank-search-input').focus(), 100);
     
     // Reset View to Initial State
     document.getElementById('bank-search-input').value = '';
     document.getElementById('bank-search-results').innerHTML = '<div class="flex flex-col items-center justify-center py-12 text-gray-400"><i class="fas fa-search-dollar text-4xl mb-3 text-gray-200"></i><p class="text-xs">Type to search bank transactions...</p></div>';
     currentBankMatches = []; 

     // Fetch data if not already loaded (Background load usually handles this)
     if (!bankDataCache) {
        loadBankData();
     }
   }

   function loadBankData() {
      // Background load - don't disturb UI unless forced
      google.script.run
        .withSuccessHandler(data => {
           if (!data.error) {
              bankDataCache = data;
              // Don't render automatically anymore, wait for search
           }
        })
        .fetchBankRecords();
   }

   function closeBankSearch() {
     document.getElementById('bank-search-modal').classList.add('hidden');
   }

   function handleBankSearch() {
     const input = document.getElementById('bank-search-input');
     let query = input.value.trim().toLowerCase();
     
     if (!bankDataCache) return;

     if (query.length === 0) {
         document.getElementById('bank-search-results').innerHTML = '<div class="flex flex-col items-center justify-center py-12 text-gray-400"><i class="fas fa-search-dollar text-4xl mb-3 text-gray-200"></i><p class="text-xs">Type to search bank transactions...</p></div>';
         return;
     }

     // NORMALIZE QUERY: Remove commas to match numbers (e.g. "1,00,000" -> "100000")
     const normalizedQuery = query.replace(/,/g, '');
     const queryWords = normalizedQuery.split(/\s+/).filter(w => w.length >= 1);
     
     const scoredMatches = bankDataCache
       .map(item => {
           if (!item.searchStr) return null;
           const searchStr = item.searchStr; // Already lowered in backend
           const searchWords = searchStr.split(/\s+/);
           let score = 0;

           queryWords.forEach(qWord => {
               // Exact match (High Priority)
               if (searchWords.some(sw => sw === qWord)) score += 10;
               // Starts with (Medium Priority)
               else if (searchWords.some(sw => sw.startsWith(qWord))) score += 5;
               // Contains (Low Priority)
               else if (searchStr.includes(qWord)) score += 3;
               // Fuzzy (Partial)
               else if (qWord.length >= 3) {
                   for (let sw of searchWords) {
                       if (sw.includes(qWord)) { score += 1; break; }
                   }
               }
           });

           return score > 0 ? { ...item, score } : null;
       })
       .filter(Boolean)
       .sort((a, b) => b.score - a.score);

     currentBankMatches = scoredMatches;
     bankDisplayLimit = BANK_PAGE_SIZE;
     renderBankSearchResults();
   }
   
   // Attach input listener for realtime search (with null check - element may not exist yet)
   (function() {
     const bankSearchInput = document.getElementById('bank-search-input');
     if (bankSearchInput) {
       bankSearchInput.addEventListener('input', handleBankSearch);
     }
   })();

   function loadMoreBankResults() {
       bankDisplayLimit += BANK_PAGE_SIZE;
       renderBankSearchResults();
   }

   function renderBankSearchResults() {
     const container = document.getElementById('bank-search-results');
     const results = currentBankMatches;
     
     if (!results || results.length === 0) {
       container.innerHTML = '<div class="flex flex-col items-center justify-center py-12 text-gray-400"><i class="fas fa-ghost text-3xl mb-2 opacity-50"></i><p class="text-xs">No matching transactions found.</p></div>';
       return;
     }

     const visibleResults = results.slice(0, bankDisplayLimit);
     
     let html = '<div class="divide-y divide-gray-100 bg-white">';
     visibleResults.forEach(tx => {
        const dateStr = new Date(tx.date).toLocaleDateString(undefined, { day: 'numeric', month: 'short', year: '2-digit' });
        const isDebit = tx.debit && tx.debit > 0;
        const isCredit = tx.credit && tx.credit > 0;
        
        // Pass FULL object to modal (encoding for safety)
        const dataStr = encodeURIComponent(JSON.stringify(tx)).replace(/'/g, "%27");

        let amountHtml = '';
        if (isDebit) amountHtml = '<span class="text-rose-600 font-bold">-' + appSettings.currency + Number(tx.debit).toLocaleString() + '</span>';
        else if (isCredit) amountHtml = '<span class="text-emerald-600 font-bold">+' + appSettings.currency + Number(tx.credit).toLocaleString() + '</span>';

        const badgeColor = tx.bank === 'SBI' ? 'bg-blue-100 text-blue-700 border-blue-200' : 'bg-green-100 text-green-700 border-green-200';
        
        html += '<div class="p-4 hover:bg-slate-50 transition-colors group cursor-pointer" onclick="openTransactionModal(\'' + dataStr + '\')">' +
           '<div class="flex justify-between items-start mb-1">' +
              '<div class="flex items-center gap-2">' +
                 '<span class="text-[9px] font-bold px-1.5 py-0.5 rounded border ' + badgeColor + '">' + tx.bank + '</span>' +
                 '<span class="text-[10px] font-bold text-gray-400">' + dateStr + '</span>' +
              '</div>' +
              amountHtml +
           '</div>' +
           '<div class="text-xs font-semibold text-slate-700 leading-normal mb-1 line-clamp-2">' + tx.description + '</div>' +
           '<div class="flex justify-between items-center text-[9px] text-gray-400">' +
              '<span class="bg-gray-100 px-1 rounded text-gray-500 font-mono">' + (tx.ref !== '-' ? extractRefDisplay(tx.ref) : '') + '</span>' +
              '<span class="font-bold text-slate-500">Bal: ' + appSettings.currency + Number(tx.balance).toLocaleString() + '</span>' +
           '</div>' +
        '</div>';
     });
     html += '</div>';
     
     if (results.length > bankDisplayLimit) {
         const remaining = results.length - bankDisplayLimit;
         html += '<div class="p-4 text-center bg-gray-50 border-t border-gray-100">' +
            '<button onclick="loadMoreBankResults()" class="text-xs font-bold text-indigo-500 hover:text-indigo-700 hover:underline transition-all">' +
               'Load More (' + remaining + ' hidden) <i class="fas fa-chevron-down ml-1"></i>' +
            '</button>' +
         '</div>';
     }
     
     container.innerHTML = html;
   }
   
   function extractRefDisplay(ref) {
       if(!ref || ref === '-') return '';
       // If ref is very long, truncate
       if(ref.length > 15) return ref.substring(0,12) + '...';
       return ref;
   }
